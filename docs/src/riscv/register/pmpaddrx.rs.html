macro_rules! reg {
    (
        $addr:expr, $csr:ident, $readf:ident, $writef:ident
    ) => {
        /// Physical memory protection address register
        pub mod $csr {
            read_csr_as_usize!($addr, $readf);
            write_csr_as_usize!($addr, $writef);
        }
    };
}
